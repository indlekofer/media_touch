# @indlekofer/media_touch

## Usage

```js
import GET_TOUCH, { REDUCER as MEDIA_REDUCER } from '@indlekofer/media_touch';

const mapStateToProps = (state) => {
  return {
    mediaTouch: state[MEDIA_REDUCER].get(GET_TOUCH)
  }
}
```
